def preprocess_text(text):
    text = text.lower() #小写
    text = re.sub(r'\d+', '', text) #删除数字
    text = re.sub(r'[^\w\s]', '', text) #删除标点符号
    words = text.split() #用空格分割
    words = [word for word in words if word not in stop_words and re.match(r'[a-zA-Z]+', word)]  # 去除停用词和非字母单词
    text = ' '.join(words)
    return text
